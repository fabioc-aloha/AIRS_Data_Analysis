% Standalone document formatting
% 1.5 line spacing for body text, single spacing for tables

\AtBeginEnvironment{longtable}{\singlespacing}

% Ensure table captions also use single spacing
\usepackage{caption}
\captionsetup[table]{font={singlespacing}}
